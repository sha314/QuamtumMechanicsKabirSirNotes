\chapter{Series} 
\section{Binomial expansion}
\begin{equation}
	(1 + x)^n = 1 + n x + \frac{n(n-1)}{2 !} x^2 + \ldots
\end{equation}

\section{Taylor expansion}
\begin{align}
	(1 - x)^{-1} = 1 + x + x^2 + x^3 + \ldots	\\
	(1 + x)^{-1} = 1 - x + x^2 - x^3 + \ldots
\end{align}
